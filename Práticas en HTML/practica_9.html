# Cules son las regiones de la OMS?

unique(covid19$Regin)
## [1] "EMRO"  "EURO"  "AFRO"  "WPRO"  "AMRO"  "SEARO" "Other"
# Se crea un factor para poder graficar las regiones (categoras)

covid19$Regin <- factor(covid19$Regin,
                         levels = c("EMRO",  "EURO",  "AFRO",  "WPRO",  "AMRO",  "SEARO", "Other"),
                         labels = c("Mediterrneo Oriental", "Europa", "frica", "Pacfico Occidental", "Amricas", "Asia Sur-Oriental", "Otros"))


# Se crea un vector con los colores a usar en el grfico:

colores <- c("#505160", "#68829E", "#AEBD38", "598234", "#1995AD", "#D9B44A", "#C99E10")


# Se grfica el factor de regiones con sus respectivas frecuencias

plot(covid19$Regin,
     main = "Casos totales de COVID-19 por regin de la OMS",
     xlab = "Regiones de la OMS",
     ylab = "Frecuencia",
     col = colores)

# Grfico de boxplot:

boxplot(covid19$Nuevos_casos ~ covid19$Regin)  # no es idonea xq hay mucha dispercin de los datos en este caso

#######################################################################

# Se filtra el Data frame en funcin de:

covid19_filtrado <- covid19 %>% 
  select(Fecha_reporte, Regin, Nuevos_casos) %>% 
  filter(Fecha_reporte == "2020-08-29") %>% 
  arrange(Nuevos_casos)


covid19_filtrado <- covid19 %>% 
  group_by(Fecha_reporte, Regin) %>% 
  summarise(Casos_acumulados = sum(Nuevos_casos))
## `summarise()` regrouping output by 'Fecha_reporte' (override with `.groups` argument)
covid19_filtrado$Fecha_reporte <- as.Date(covid19_filtrado$Fecha_reporte, format = "%Y-%m-%d")


class(covid19_filtrado$Fecha_reporte)
## [1] "Date"
class(covid19_filtrado$Regin)
## [1] "factor"
class(covid19_filtrado$Casos_acumulados)
## [1] "integer"

# ggplot(data, aes(x = variable explicativa, y = var. dependiente, fill =))   aesthetic specifications

# Diagrama de cajas:

options(scipen = 100000000) # funcin para prevenir notacin cientfica:

ggplot(covid19_filtrado, aes(x = Fecha_reporte, 
                             y = Casos_acumulados,
                             fill = Regin)) +
                              geom_boxplot() + 
                              ggtitle("Boxplot de casos acumulaods en funcin de la fecha de reporte por regin OMS") +
                              labs( x = "Fecha de reporte", y = "Casos acumulados")

# Diagrama de Lneas

ggplot(covid19_filtrado, aes(x = Fecha_reporte, 
                             y = Casos_acumulados,
                             col = Regin)) + 
                              geom_line() +
                             ggtitle("Densidad de casos acumulaods en funcin de la fecha de reporte por regin OMS") +
                              labs( x = "Fecha de reporte", 
                                    y = "Casos acumulados")

# Diagrama de puntos 

ggplot(covid19_filtrado, aes(x = Fecha_reporte, 
                             y = Casos_acumulados,

                              ggtitle("Dispersin de los casos acumulaods en funcin de la fecha de reporte por regin OMS") +
